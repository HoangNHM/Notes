#Java
1.
mvn archetype:generate -DgroupId=com.mycompany.app -DartifactId=my-app -DarchetypeArtifactId=maven-archetype-quickstart -DinteractiveMode=false

2.
cd my-app

3.
mvn package

4.
java -cp target/my-app-1.0-SNAPSHOT.jar com.mycompany.app.App

5.
mvn site
